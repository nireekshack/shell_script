#!/bin/bash

echo "*** multiplication in static ***"
a=10
b=10
c=$((a*b))
echo "multiplication of $a and $b : $c"

echo "---------------------------------"

echo "*** multiplication in dynamic ***"
echo "enter value of d"
read d
echo "enter value of e"
read e
f=$((d*e))
echo "multiplication of $d and $e : $f"
